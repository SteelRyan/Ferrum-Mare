country_event = {
	id = 4760086
	title = "The Crown of Teril"
	desc = "4760086_desc"
	major = yes

	trigger = { has_country_modifier = dissolve_teril_event }
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {

	}

	option = {
		name = "Endorse Teril's dissolution"
		remove_country_modifier = dissolve_teril_event

	}
	option = {
		name = "Defend Teril's sovereignty"
		remove_country_modifier = dissolve_teril_event
		BOT = {
			change_variable = {
				which = TER_score
				value = -1.000
			}
		}	 
	}
}

country_event = {
	id = 4760087
	title = "Total Opposition to Teril's Dissolution"
	desc = "The proposal to dissolve the legal status of Teril has been rejected by all relevant powers, preserving its nominal existence despite its lack of true sovereignty."
	major = yes

	trigger = {
		OR = {
			tag = RE1
			tag = EMA
			tag = DEV
		}
		RE1 = { NOT = { has_country_modifier = dissolve_teril_event } }
		EMA = { NOT = { has_country_modifier = dissolve_teril_event } }
		DEV = { NOT = { has_country_modifier = dissolve_teril_event } }
		NOT = { has_country_flag = dissolve_teril_resolved }
		BOT = { 
			has_country_flag = dissolve_teril 
			NOT = { 
				check_variable = {
					which = TER_score
					value = 1
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_country_flag = dissolve_teril_resolved
	}

	option = {
		name = "Teril’s status remains"

	}
}
